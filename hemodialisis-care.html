// Profile editing functions

function editProfile() {

document.getElementById('profile-display').style.display = 'none';

document.getElementById('profile-form').style.display = 'block';

}

 

function saveProfile() {

const name = document.getElementById('name-input').value;

const age = document.getElementById('age-input').value;

const weight = document.getElementById('weight-input').value;

// Update display

document.querySelector('#profile-display p:nth-child(1)').innerHTML = `<strong>Nama:</strong> ${name}`;

document.querySelector('#profile-display p:nth-child(2)').innerHTML = `<strong>Umur:</strong> ${age} tahun`;

document.querySelector('#profile-display p:nth-child(3)').innerHTML = `<strong>Berat Badan:</strong> ${weight} kg`;

// Hide form and show updated profile

document.getElementById('profile-form').style.display = 'none';

document.getElementById('profile-display').style.display = 'block';

// Show success message

alert('Data pasien berhasil diperbarui!');

}

 

function cancelEdit() {

document.getElementById('profile-form').style.display = 'none';

document.getElementById('profile-display').style.display = 'block';

}

 

// Simulate notifications every hour

const messages = [

"Waktu minum air!",

"Persiapan untuk dialisis besok"

];

const randomMessage = messages[Math.floor(Math.random() * messages.length)];

// Create toast notification

const toastContainer = document.createElement('div');

toastContainer.className = 'position-fixed bottom-0 end-0 p-3';

toastContainer.style.zIndex = '11';

const toastEl = document.createElement('div');

toastEl.className = 'toast show';

<div class="toast-header">

<strong class="me-auto">Pengingat</strong>

<small class="text-muted">Baru saja</small>

<button type="button" class="btn-close" data-bs-dismiss="toast"></button>

</div>

<div class="toast-body">${randomMessage}</div>

`;

toastContainer.appendChild(toastEl);

document.body.appendChild(toastContainer);

setTimeout(() => {

toastContainer.remove();

}, 5000);

}

}, 3600000); // Every hour
